function int fib(int n) {

    var int v1, v2, ret;

    if(n <= 1) {
        ret = n;
    } else {
        v1 = fib(n-1);
        v2 = fib(n-2);
        ret = v1 + v2;
    };

    return (ret);
}

function int fib2(int n) {

    var int ret;

    if(n <= 1) {
        ret = n;

        ret = fib2(n-1) + fib2(n-2);

}
