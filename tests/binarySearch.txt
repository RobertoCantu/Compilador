program binarySearch_;

var int arr[5], i, res;

functions
function int binary(int l, int r, int x){
  var int mid, retVal;
  
  if(r >= l){
    mid = l + (r - l) / 2;

    if(arr[mid] == x){
      retVal = mid;
    } else {
      if(arr[mid] > x){
        retVal = binary(l, mid - 1, x);
      } else {
        retVal = binary(mid + 1, r, x);
      };
    };

    
  } else {
    retVal = 0;
  };

  return (retVal);

}
