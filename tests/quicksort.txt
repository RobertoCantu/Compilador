program test_;

var int arr[6], x, i;


functions
function int partition(int low, int high) {
    var int pivot, i, aux, aux2, j;

    pivot = arr[high];
    i = (low - 1);

    j = low;

    while(j < high) {

        if (arr[j] < pivot) {

            i = i + 1;

            aux = arr[i];
            aux2 = arr[j];

            arr[i] = aux2;
            arr[j] = aux;
        };

        j = j + 1;
    };

    aux = arr[i + 1];
    aux2 = arr[high];

    arr[i + 1] = aux2;
    arr[high] = aux;

    return (i + 1);
}


function void quickSort(int low, int high) {
    var int pi;

    if (low < high) {

        pi = partition(low, high);
 
        quickSort(low, pi - 1);
        quickSort(pi + 1, high);
    };

}

function void printArray() {
    var int i;

    write("here");

    for i = 0 to 6 do {
        write(arr[i]);
    }; 
}


main {
